-- Database: Knowlage_Base

-- DROP DATABASE IF EXISTS "Knowlage_Base";

CREATE DATABASE "Knowlage_Base"
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'Russian_Russia.1251'
    LC_CTYPE = 'Russian_Russia.1251'
    LOCALE_PROVIDER = 'libc'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;


-- Table: public.tags

-- DROP TABLE IF EXISTS public.tags;

CREATE TABLE IF NOT EXISTS public.tags
(
    tg_id integer NOT NULL,
    tg_name text COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT tags_pkey PRIMARY KEY (tg_id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.tags
    OWNER to postgres;

-- Table: public.topics

-- DROP TABLE IF EXISTS public.topics;

CREATE TABLE IF NOT EXISTS public.topics
(
    t_id integer NOT NULL,
    t_name text COLLATE pg_catalog."default",
    CONSTRAINT topics_pkey PRIMARY KEY (t_id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.topics
    OWNER to postgres;


-- Table: public.terms

-- DROP TABLE IF EXISTS public.terms;

CREATE TABLE IF NOT EXISTS public.terms
(
    term_id integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    term_topic integer NOT NULL,
    term_tag integer NOT NULL,
    term_name text COLLATE pg_catalog."default" NOT NULL,
    term_lemma text COLLATE pg_catalog."default" NOT NULL,
    term_res text COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT terms_pkey PRIMARY KEY (term_id),
    CONSTRAINT fk_tag FOREIGN KEY (term_tag)
        REFERENCES public.tags (tg_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT fk_topic FOREIGN KEY (term_topic)
        REFERENCES public.topics (t_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.terms
    OWNER to postgres;
